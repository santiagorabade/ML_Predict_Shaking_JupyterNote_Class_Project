auX = df[['Ml','epicentral_distance']]
X = np.array(auX)
auY = df[['channel_pgv_ms']]
y = np.array(auY)
y = np.log10(y)
print("X's shape is:", X.shape)
print("y's shape is:", y.shape)
newY=np.zeros(302)
newX=np.zeros((302,2))
counter=0
#print(indexPos)
for x in range(0, len(df), 2):
    if (y[x]>y[x+1]):
        newY[counter]=y[x]
        newX[counter]=X[x]
    else:
        newY[counter]=y[x+1]
        newX[counter]=X[x+1]  
    counter+=1
